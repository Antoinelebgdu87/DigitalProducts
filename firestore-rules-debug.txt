// Règles Firestore temporaires pour déboguer - À COPIER dans Firebase Console
// ⚠️ CES RÈGLES SONT OUVERTES ET PAS SÉCURISÉES - POUR DEBUG SEULEMENT

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Permettre toutes les lectures et écritures temporairement
    match /{document=**} {
      allow read, write: if true;
    }
  }
}
